// O(N + M)
const int MAX = 2*MAXN+10;
 
int n, m, timer = 0;
int ns[MAX];
vector<vector<int>> adj;
vector<int> tin, tout;
 
void dfs(int u, int p) {
  tin[u] = ++timer;
  for(int v : adj[u]) {
    if(v == p) continue;
    dfs(v, u);
  }
  tout[u] = ++timer;
}
