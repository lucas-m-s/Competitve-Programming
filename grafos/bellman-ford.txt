//O(V*E)
struct Edge {int a, b, w;};
int n, m;//n=|V|, m=|E|

vector<int> bellmanford(int s, vector<Edge> &edges)
{
    vector<int> dist(n, INF);//0-based indexing
    dist[s] = 0;//s é o vértice de início
    bool neg;
    for(int i = 0; i < n; i++){
        neg = 0;
        for(auto e : edges){
            if(dist[e.a] != INF and dist[e.b] > dist[e.a] + e.w){
                dist[e.b] = dist[e.a] + e.w;
                neg = 1;
            }
        }
    }
    if(neg) cout << "Tem ciclo negativo\n";
    return dist;
}
